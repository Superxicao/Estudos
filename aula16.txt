lanche=('Hamburguer','Suco','Pizza','Pudim') # pode escrever sem parênteses tambem
lanche='Hamburguer','Suco','Pizza','Pudim')
print(lanche)		ele mostra a tupla
print(lanche[1])	ele mostra o segundo elemento da tupla, no caso o suco
print(lanche[3])	ele mostra o quarto elemento, que é o pudim
print(lanche[-2])	ele mostra o elemento pizza, que é o elemento 2 (terceiro elemento) ou elemento -2
print(lanche[1:3])	ele mostra o suco, pizza sem o ultimo elemento q é o limite aonde ele vai parar sem incluir, elementos 1 e 2 ele mostra
print(lanche[2:])	ele mostra do elemento 2 (terceiro elemento) até o último, no caso pizza e pudim
print(lanche[:2])	ele mostra do início até o elemento 2 (terceiro elemento) sem mostrar o terceiro elemento, que é o elemento 2, então ele mostra hamburguer e suco
print(lanche[-2:)	ele mostra da pizza até fim, no caso, pizza e pudim, do -2 pra frente
print(lanche[-3:)	ele mostra o suco pizza e pudim, do -3 q tb é o elemento 1 (segundo elemento) até o fim, mostrando suco, pizza e pudim
